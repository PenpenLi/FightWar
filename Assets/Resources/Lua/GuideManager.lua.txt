end

function SetButtonLoaclPosition(v3)

end

end

function SetGuiderLocalPosition( PositionV3,RotationV3,TextV3)
	GameObject.Find("Guider").transform.localPosition=PositionV3
	GameObject.Find("Guider").transform.localRotation=RotationV3
	--GameObject.Find("Guider").transform.eulerAngles=RotationV3
	GameObject.Find("GuideText").transform.localPosition=TextV3
	--GameObject.Find("GuideText").transform.eulerAngles=RotationV3
	GameObject.Find("GuideText").transform.localRotation=RotationV3
end

function SetGuiderText( text )
	if GameObject.Find("GuideText") ~= nil then
		textLabel = GameObject.Find("GuideText"):GetComponent("UILabel")
		textLabel.text=text
	end
end
